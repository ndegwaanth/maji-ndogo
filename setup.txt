MYSQL SETUP IN GIT BASH
1.DOWNLOAD ZIP FILE
curl -L -o mysql-noadmin.zip https://dev.mysql.com/get/Downloads/MySQL-8.0/mysql-8.0.40-winx64.zip

Step 2: Extract the ZIP File
mkdir -p ~/mysql
unzip mysql-noadmin.zip -d ~/mysql

Step 3: Initialize the Data Directory
cd ~/mysql/mysql-8.0.40-winx64
./bin/mysqld --initialize-insecure --basedir=$(pwd) --datadir=$(pwd)/data

Step 4: Start MySQL Server (User Mode)
./bin/mysqld --console --datadir=$(pwd)/data

Step 5: Connect to MySQL
cd ~/mysql/mysql-8.0.40-winx64
./bin/mysql -u root

OPEN ANOTHER GIT BASH SHELL AND CONTINUE WITH THE REST OF THE COMMAND

Step 6: Add MySQL to PATH (optional)
If you want to use mysql from anywhere:
echo 'export PATH=$PATH:"/c/Users/Anthony/mysql/mysql-8.0.40-winx64/bin"' >> ~/.bashrc
source ~/.bashrc
